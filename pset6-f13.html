# finite-difference (FD) Green's function:
N=100
L = 1
dx = L/(N+1)
D = diff1(N)/dx
function A()
  return D'*D - ^2*eye(N)
end
function d(x)
  k = iround(x/L * (N+1))
  k = k < 1 ? 1 : (k > N ? N : k)
  dk = zeros(N)
  dk[k] = 1/dx
  return dk
end
g(x,) = A() \ d(x)

# analytical Green's function:
G(x,y,) = (x < y ? sin(*x)*sin(*(L-y)) : sin(*y)*sin(*(L-x))) / ( * sin(*L))
G(X::Array,y,) = [ G(x,y,) for x in X ]

Hooray, we have a pretty good match between the exact \(G(x,x')\) Green's function and the finite-difference (FD) approximation. At the higher frequency, the match is not quite as good, but that isn't too surprising: -the difference approximations become less and less accurate as the function becomes more and more oscillatory relative to the grid \(\Delta x\).
